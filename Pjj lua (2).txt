local function GetNearestEnemy()
    local char = LP.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then return nil end

    local nearest = nil
    local minDistance = 500

    for _, v in pairs(Workspace:GetChildren()) do
        if v:IsA("Model") and v ~= char and v:FindFirstChild("Humanoid") and v:FindFirstChild("HumanoidRootPart") then
            if v.Humanoid.Health > 0 then
                local dist = (root.Position - v.HumanoidRootPart.Position).Magnitude
                if dist < minDistance then 
                    minDistance = dist
                    nearest = v
                end
            end
        end
    end
    return nearest
end

-- Chống kẹt tọa độ chuột trên điện thoại
local function getSafeAim()
    local mouse = LP:GetMouse()
    if mouse.Hit and mouse.Hit.Position ~= Vector3.new(0,0,0) then 
        return mouse.Hit 
    end
    local char = LP.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        return char.HumanoidRootPart.CFrame * CFrame.new(0, 0, -50)
    end
    return CFrame.new()
end

local function fireSkillRemote(letter)
    local remotes = skillMappings[letter]
    if remotes then
        local hitCFrame = getSafeAim()
        for _, name in ipairs(remotes) do
            local r = ReplicatedStorage:FindFirstChild(name, true) or ReplicatedStorage:FindFirstChild(name:gsub("^%l", string.upper), true)
            if r and r:IsA("RemoteEvent") then
                pcall(function() 
                    r:FireServer(true) 
                    r:FireServer(hitCFrame) 
                    r:FireServer() 
                end)
            end
        end
    end
end

local function getHitboxTarget(hum, v)
    if not hum or hum.Health <= 0 or v.Name == LP.Name then return false end
    if Players:FindFirstChild(v.Name) then
        if not v:FindFirstChildOfClass("ForceField") then
            if flags.killAll then return true end
            if flags.farmPlayer and flags.targetPlayer ~= "" and string.find(string.lower(v.Name), string.lower(flags.targetPlayer)) then
                return true
            end
        end
        return false 
    end
    if flags.customBoss ~= "" and string.find(string.lower(v.Name), string.lower(flags.customBoss)) then return true end
    if flags.boss == v.Name then return true end
    if flags.afm and not v:FindFirstChildOfClass("ForceField") then return true end
    if flags.mfi and (v.Name == "Space Dummy" or v.Name == "Holy Dummy") then return true end
    if flags.mfm and (v.Name == "Metal Dummy" or v.Name == "Wood Dummy" or v.Name == "Ice Dummy" or v.Name == "Rock Dummy") then return true end
    return false
end

local function getClosestPlayer()
    local closestDist = 150 
    local closestTarget = nil
    local char = LP.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    for _, v in pairs(Players:GetPlayers()) do
        if v ~= LP and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
            local hum = v.Character:FindFirstChild("Humanoid")
            if hum and hum.Health > 0 then
                local dist = (v.Character.HumanoidRootPart.Position - char.HumanoidRootPart.Position).Magnitude
                if dist < closestDist then
                    closestDist = dist
                    closestTarget = v.Character.HumanoidRootPart
                end
            end
        end
    end
    return closestTarget
end

local function getTargetPlayerByName()
    if flags.targetPlayer == "" then return nil end
    local char = LP.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    for _, v in pairs(Players:GetPlayers()) do
        if v ~= LP and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
            if string.find(string.lower(v.Name), string.lower(flags.targetPlayer)) or string.find(string.lower(v.DisplayName), string.lower(flags.targetPlayer)) then
                local hum = v.Character:FindFirstChild("Humanoid")
                if hum and hum.Health > 0 then return v.Character.HumanoidRootPart end
            end
        end
    end
    return nil
end

                local badAttributes = {"stunned", "epitaphSilence", "RealStun", "B2m_Knocked_Out", "GrabStun", "echoesBarrage", "echoesR", "echoesGravitySub", "B2m_Slow_Debuff"}
                for _, attr in ipairs(badAttributes) do
                    if char:GetAttribute(attr) then char:SetAttribute(attr, false) end
                end
                
                local badVals = {"UltSpinDebuff", "roomvar", "silenced", "metallicasilenced", "uMoveStunned", "poseStunned", "weakened"}
                for _, vName in ipairs(badVals) do
                    local val = char:FindFirstChild(vName) or LP:FindFirstChild(vName)
                    if val and val:IsA("ValueBase") then 
                        if typeof(val.Value) == "boolean" and val.Value == true then
                            val.Value = false
                        elseif typeof(val.Value) == "number" and val.Value > 0 then
                            val.Value = 0
                        end
                    end
                end
                
                -- TIÊU DIỆT HẠT HỮU CƠ (BLEED, VENOM, FIRE, SLEEP) TỪ ĐẦU TỚI CHÂN
                for _, obj in pairs(char:GetDescendants()) do
                    if obj:IsA("ParticleEmitter") or obj:IsA("Sound") or obj:IsA("BillboardGui") then
                        local en = obj.Name:lower()
                        if en:match("bleedparticle") or en:match("venomparticle") or en:match("flameeffect") or en:match("burneffect") or en:match("sleepzs") or en:match("snore") or en:match("music") or en:match("hiteffect") or en:match("impact") then
                            obj:Destroy()
                        end
                    end
                end

                for _, v in pairs(char:GetChildren()) do
                    local n = v.Name:lower()
                    if n:match("timestop") or n:match("slow") or n:match("hide") or n:match("blind") or n:match("disable") or n:match("weld") or n:match("grab") or n:match("harvest") then
                        v:Destroy()
                    end
                end
                
                if workspace:FindFirstChild("FakeWeldParts") then
                    for _, v in pairs(workspace.FakeWeldParts:GetChildren()) do
                        if v.Name:lower():match("bubble") then v:Destroy() end
                    end
                end
                if workspace:FindFirstChild("Effects") then
                    for _, v in pairs(workspace.Effects:GetChildren()) do
                        local n = v.Name:lower()
                        if n:match("wamuubleed") or n:match("skinpop") or n:match("spikeboom") then v:Destroy() end
                    end
                end

                local hrp = char:FindFirstChild("HumanoidRootPart")
                if hrp and hrp.Anchored then hrp.Anchored = false end

                for _, obj in pairs(char:GetChildren()) do
                    if obj:IsA("BasePart") and obj ~= hrp then
                        local n = obj.Name:lower()
                        if n:match("ice") or n:match("stun") or n:match("freeze") or n:match("trap") or n:match("bubble") or n:match("rock") then obj:Destroy() end
                    end
                end

                -- TRỤC XUẤT UI RÁC CỦA TIMESTOP VÀ CRAZY TRAIN
                local pg = LP:FindFirstChild("PlayerGui")
                if pg then
                    for _, v in pairs(pg:GetChildren()) do
                        local n = v.Name:lower()
                        if n:match("blind") or n:match("hide") or n:match("black") or n:match("flash") or n:match("drunk") or n:match("wou") or n:match("bubble") or n:match("pop") or n:match("harvestgui") or n:match("charge") or n:match("crazy") or n:match("train") or n:match("timestop") or n:match("tsgui") or n:match("color") or n:match("green") then 
                            v:Destroy() 
                        end
                    end
                end
            end)
        end
    end
end)

RunService.Stepped:Connect(function()
    local char = LP.Character
    if char and flags.autoKite then
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hrp then
            pcall(function()
                local closestThreat = nil
                local minDist = 25 
                for _, v in pairs(Workspace:GetChildren()) do
                    if v ~= char and v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") then
                        if not Players:GetPlayerFromCharacter(v) and not v.Name:match("Dummy") then 
                            local enemyHum = v:FindFirstChild("Humanoid")
                            if enemyHum.Health > 0 then 
                                local dist = (hrp.Position - v.HumanoidRootPart.Position).Magnitude
                                if dist < minDist then minDist = dist closestThreat = v.HumanoidRootPart end
                            end
                        end
                    end
                end
                
                if closestThreat then
                    local direction = (hrp.Position - closestThreat.Position).Unit
                    local pushBack = Vector3.new(direction.X, 0, direction.Z).Unit * 1.2
                    hrp.CFrame = hrp.CFrame + pushBack
                end
            end)
        end
    end
end)
